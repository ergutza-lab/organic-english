// if (action === 'getDatosCA') return getDatosCA(); // if (action === 'getAsistenciaFecha') return getAsistenciaFecha(e.parameter.fecha); // ============================================================================= // loginCA — Login del coordinador académico // Valida ID + clave en pestaña CONTROL_ACADEMICO (col A = ID, col D = Clave) // ============================================================================= function loginCA(id, clave) { try { var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = ss.getSheetByName('CONTROL_ACADEMICO'); var datos = sheet.getDataRange().getValues(); for (var i = 1; i < datos.length; i++) { if (datos[i][0] === id && String(datos[i][3]) === String(clave)) { return ContentService .createTextOutput(JSON.stringify({ status: 'ok', id: datos[i][0], nombre: datos[i][1] })) .setMimeType(ContentService.MimeType.JSON); } } return ContentService .createTextOutput(JSON.stringify({ status: 'error', mensaje: 'ID o clave incorrectos' })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', mensaje: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ============================================================================= // getDatosCA — Datos completos para el panel de Control Académico // Regresa: maestros, resumen semanal, clases hoy y clases esta semana // ============================================================================= function getDatosCA() { try { var ss = SpreadsheetApp.openById(SHEET_ID); // ── Cargar maestros ────────────────────────────────────────────── var sheetMaestros = ss.getSheetByName('MAESTROS'); var filasMaestros = sheetMaestros.getDataRange().getValues(); var maestros = []; for (var i = 1; i < filasMaestros.length; i++) { var m = filasMaestros[i]; if (!m[0]) continue; maestros.push({ id: m[0], // col A nombre: m[1], // col B whatsapp: String(m[2]).replace(/\s+/g, ''), // col C email: m[3], // col D rol: m[4], // col E activo: m[5] === true || String(m[5]).toUpperCase() === 'TRUE', // col F foto: m[7] || '', // col H grupos: m[8] || 0 // col I (fórmula combinada GRUPOS col L + col O) }); } // ── Calcular clases hoy y esta semana desde ASISTENCIA_MAESTROS ── var hoy = new Date(); var hoyStr = Utilities.formatDate(hoy, 'America/Mexico_City', 'yyyy-MM-dd'); var diaSemana = hoy.getDay(); var inicioSemana = new Date(hoy); inicioSemana.setDate(hoy.getDate() - (diaSemana === 0 ? 6 : diaSemana - 1)); inicioSemana.setHours(0, 0, 0, 0); var sheetAsist = ss.getSheetByName('ASISTENCIA_MAESTROS'); var clasesHoy = 0; var clasesSemana = 0; var clasesPorMaestro = {}; // { idMaestro: { dadas: N, esperadas: N } } // Inicializar contadores por maestro activo maestros.filter(function(m) { return m.activo; }).forEach(function(m) { clasesPorMaestro[m.id] = { dadas: 0, esperadas: 0 }; }); if (sheetAsist) { var filasAsist = sheetAsist.getDataRange().getValues(); for (var a = 1; a < filasAsist.length; a++) { var fa = filasAsist[a]; var idMaest = fa[0]; // col A var fechaA = Utilities.formatDate(new Date(fa[2]), 'America/Mexico_City', 'yyyy-MM-dd'); // col C var dioclase = fa[3] === true || String(fa[3]).toUpperCase() === 'TRUE'; // col D var fechaObj = new Date(fa[2]); var enSemana = fechaObj >= inicioSemana && fechaObj <= hoy; if (enSemana && clasesPorMaestro[idMaest] !== undefined) { clasesPorMaestro[idMaest].esperadas++; if (dioclase) { clasesPorMaestro[idMaest].dadas++; clasesSemana++; if (fechaA === hoyStr) clasesHoy++; } } } } // Armar resumen semanal var resumenSemana = maestros .filter(function(m) { return m.activo; }) .map(function(m) { var c = clasesPorMaestro[m.id] || { dadas: 0, esperadas: 0 }; return { id: m.id, nombre: m.nombre, dadas: c.dadas, esperadas: c.esperadas }; }); return ContentService .createTextOutput(JSON.stringify({ status: 'ok', maestros: maestros, clasesHoy: clasesHoy, clasesSemana: clasesSemana, resumenSemana: resumenSemana })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', mensaje: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ============================================================================= // getAsistenciaFecha — Regresa los grupos activos para una fecha dada // Usado para cargar la tabla de asistencia del día seleccionado // ============================================================================= function getAsistenciaFecha(fecha) { try { var ss = SpreadsheetApp.openById(SHEET_ID); // Cargar grupos con maestro asignado (col L de GRUPOS) var sheetGrupos = ss.getSheetByName('GRUPOS'); var grupos = sheetGrupos.getDataRange().getValues(); // Cargar asistencia ya registrada para esta fecha var sheetAsist = ss.getSheetByName('ASISTENCIA_MAESTROS'); var asistencias = sheetAsist ? sheetAsist.getDataRange().getValues() : []; // Índice de asistencias ya registradas: "idMaestro_idGrupo_fecha" → dioclase var asistIndex = {}; for (var a = 1; a < asistencias.length; a++) { var fa = asistencias[a]; var fechaA = Utilities.formatDate(new Date(fa[2]), 'America/Mexico_City', 'yyyy-MM-dd'); if (fechaA === fecha) { var key = fa[0] + '_' + fa[1] + '_' + fechaA; asistIndex[key] = fa[3] === true || String(fa[3]).toUpperCase() === 'TRUE'; } } // Cargar maestros para obtener nombres var sheetMaestros = ss.getSheetByName('MAESTROS'); var maestros = sheetMaestros.getDataRange().getValues(); var nombresMaestros = {}; for (var m = 1; m < maestros.length; m++) { nombresMaestros[maestros[m][0]] = maestros[m][1]; } // Filtrar grupos con maestro asignado y estado DISPONIBLE o con alumnos var rows = []; for (var g = 1; g < grupos.length; g++) { var idMaestro = grupos[g][11]; // col L if (!idMaestro) continue; var idGrupo = grupos[g][0]; var key = idMaestro + '_' + idGrupo + '_' + fecha; rows.push({ idGrupo: idGrupo, horario: grupos[g][2], nivel: grupos[g][4], maestroId: idMaestro, maestroNombre: nombresMaestros[idMaestro] || idMaestro, dioclase: asistIndex[key] !== undefined ? asistIndex[key] : false }); } return ContentService .createTextOutput(JSON.stringify({ status: 'ok', rows: rows })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', mensaje: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ============================================================================= // guardarAsistencia — Guarda o actualiza registros en ASISTENCIA_MAESTROS // Recibe arreglo de { idMaestro, idGrupo, fecha, dioclase, idCA } // ============================================================================= function guardarAsistencia(registros) { try { var ss = SpreadsheetApp.openById(SHEET_ID); var sheetAsist = ss.getSheetByName('ASISTENCIA_MAESTROS'); if (!sheetAsist) { // Crear la pestaña si no existe sheetAsist = ss.insertSheet('ASISTENCIA_MAESTROS'); sheetAsist.appendRow(['ID_Maestro','ID_Grupo','Fecha','Dio_Clase','ID_Coordinador']); } var datos = sheetAsist.getDataRange().getValues(); registros.forEach(function(r) { var encontrado = false; for (var i = 1; i < datos.length; i++) { var fechaFila = Utilities.formatDate(new Date(datos[i][2]), 'America/Mexico_City', 'yyyy-MM-dd'); if (datos[i][0] === r.idMaestro && datos[i][1] === r.idGrupo && fechaFila === r.fecha) { // Actualizar fila existente sheetAsist.getRange(i + 1, 4).setValue(r.dioclase); sheetAsist.getRange(i + 1, 5).setValue(r.idCA); encontrado = true; break; } } if (!encontrado) { sheetAsist.appendRow([r.idMaestro, r.idGrupo, r.fecha, r.dioclase, r.idCA]); } }); return ContentService .createTextOutput(JSON.stringify({ status: 'ok' })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', mensaje: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ============================================================================= // altaMaestro — Registra un maestro nuevo en MAESTROS // Genera ID automático (MAE-001, MAE-002...) // ============================================================================= function altaMaestro(datos) { try { var ss = SpreadsheetApp.openById(SHEET_ID); var sheet = ss.getSheetByName('MAESTROS'); var filas = sheet.getDataRange().getValues(); // Generar siguiente ID var ultimoNum = 0; for (var i = 1; i < filas.length; i++) { var id = String(filas[i][0]); return ContentService .createTextOutput(JSON.stringify({ status: 'ok', idMaestro: idMaestro })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: 'error', mensaje: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ============================================================================= // bajaMaestro — Cambia Activo a FALSE en MAESTROS col F // Los grupos se reasignan manualmente en la hoja GRUPOS // ============================================================================= function bajaMaestro(idMaestro) { try { var ss = SpreadsheetApp.openById(SHEET_ID);
